\newcommand\setup[0]{
	% grid
	\draw[step=1.0,black,thin] (0,0) grid (10,9);
	% ticks
	\foreach \x in {0,...,9} {
		\node[align=left] at (\x + 0.5,-0.5) {\x};
	}
	% axis
	\draw [thick, |->] (0,-1) -- (9.5,-1);
	\node[align=left] at (9,-1.5) {x};
}
